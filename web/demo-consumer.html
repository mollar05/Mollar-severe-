<!-- 1) TF.js --> <script src="https://cdn.jsdelivr.net/npm/@tensorflow/tfjs@4.15.0/dist/tf.min.js"></script> <!-- 2) TeachableTextAI 라이브러리 --> <script src="teachable-text-ai.js"></script> <script> // 3) 모델 URL로 로드 (서버에 올린 경우) await TeachableTextAI.load( 'https://your-server.com/model/model.json', 'https://your-server.com/model/weights.bin', { onProgress: (pct, msg) => console.log(pct + '% ' + msg) } ); // 4) 예측 const result = await TeachableTextAI.predict("안녕하세요"); console.log(result.label); // "인사" console.log(result.confidence); // 0.97 console.log(result.scores); // { 인사: 0.97, 음식: 0.03 } </script>
<!-- HTML --> <input type="file" id="jsonFile" accept=".json"> <input type="file" id="binFile" accept=".bin"> <button onclick="load()">모델 불러오기</button> <!-- JS --> async function load() { const jsonFile = document.getElementById('jsonFile').files[0]; const binFile = document.getElementById('binFile').files[0]; await TeachableTextAI.loadFromFiles(jsonFile, binFile, { onProgress: (pct, msg) => console.log(pct + '%', msg) }); const result = await TeachableTextAI.predict("치킨 먹자"); console.log(result); // { label: "음식", confidence: 0.99, scores: {...} } }
// onReady 콜백 패턴 — 로드 완료 시 자동 실행 TeachableTextAI.onReady(({ labels }) => { console.log('사용 가능한 클래스:', labels); }); // isReady() 로 즉시 확인 후 사용 if (TeachableTextAI.isReady()) { const { label, confidence } = await TeachableTextAI.predict(userInput); showResult(label, confidence); } // getLabels() — 클래스 목록 가져오기 const classes = TeachableTextAI.getLabels(); // ["인사", "음식", ...] // unload() — 메모리 해제 TeachableTextAI.unload();
